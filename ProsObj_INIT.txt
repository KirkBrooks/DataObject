This allows me to use  // this method on-the-fly and not over-write anything already defined.  // Default error handling:C_OBJECT(prosObject)Case of: (OB Is defined(prosObject)): (Not(ProsObj_LOCK_SET ))ElseprosObject:=OB_newOBJ_Set_Text (prosObject;"startedAt";String(Current date;ISO date;Current time))OBJ_Set_Text (prosObject;"user-4D";Current user)OBJ_Set_Text (prosObject;"process.name";Pros_get_name )OBJ_Set_Long (prosObject;"process.number";Current process)OBJ_Set_Text (prosObject;"process.UUID";Lowercase(Generate UUID))ProsObj_LOCK_CLEAREnd case  //[End ProsObj_INIT ]
